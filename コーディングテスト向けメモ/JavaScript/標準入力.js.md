
### １行目に後続の行数Nなどが指定されるパターン ex."N"

~~~js
var lines = require("fs").readFileSync("/dev/stdin", "utf8").split("\n");
const N = lines[0];
for(var i = 0; i < N; i++) {
  var line = lines[i+1].split(" ");
}
~~~

### １行目に後続の行数Nとその他条件の整数Mなどが並ぶパターン ex."N M"

~~~js
var lines = require("fs").readFileSync("/dev/stdin", "utf8").split("\n");
var NM = lines[0].split(" ");
const N = NM[0];
const M = NM[1];
for(var i = 0; i < N; i++) {
  var line = lines[i+1].split(" ");
}
~~~
